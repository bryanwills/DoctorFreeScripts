659 function rawurlencode() {
660  local string="${1}"
661  local strlen=${#string}
662  local encoded=""
663 
664  for (( pos=0 ; pos<strlen ; pos++ )); do
665  c=${string:$pos:1}
666  case "$c" in
667  [-_.~a-zA-Z0-9] ) o="${c}" ;;
668  * ) printf -v o '%%%02x' "'$c"
669  esac
670  encoded+="${o}"
671  done
672  echo "${encoded}"
673 }
