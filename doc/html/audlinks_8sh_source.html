74 ## @fn FindAndLink()
75 ## @brief Remove file and replace with symbolic link
76 ## @param param1 first argument is directory path in which to find files
77 FindAndLink() {
78  T="$1"
79  abook=
80  [ "$T" = "$A" ] && {
81  abook=1
82  # iTunes doesn't use the audiobook title in its library hierarchy
83  title=
84  }
85  find . -type f | while read i
86  do
87  found=
88  if [ -L "$i" ]
89  then
90  continue
91  else
92  b=`basename "$i"`
93  echo "$b" | grep ":" > /dev/null && {
94  b=`echo "$b" | sed -e "s/:/_/g"`
95  }
96  d=`dirname "$i"`
97  [ "$abook" ] || {
98  title=`basename "$d"`
99  echo "$title" | grep ":" > /dev/null && {
100  title=`echo "$title" | sed -e "s/:/_/g"`
101  }
102  }
103  e=`dirname "$d"`
104  artist=`basename "$e"`
105  [ "$b" = "SUMS" ] && continue
106  [ "$USE_SUM" ] && {
107  ck1=`grep "$i" SUMS | awk ' { print $1 } '`
108  [ "$ck1" ] || ck1=`cksum "$i" | awk ' { print $1 } '`
109  }
110  [ -f "$T/$artist/$title/$b" ] && found="$b"
111  [ "$found" ] || {
112  # Check for file with track number added as prefix
113  [ -d "$T/$artist/$title" ] && {
114  pushd "$T/$artist/$title" > /dev/null
115  for song in [0-9][0-9]*
116  do
117  notrack=${song:3}
118  if [ "$b" = "$notrack" ]
119  then
120  found="$song"
121  break
122  fi
123  done
124  popd > /dev/null
125  }
126  }
127  [ "$found" ] && {
128  if [ "$USE_SUM" ]
129  then
130  ck2 =`grep "$found" "$T/SUMS" | awk ' { print $1 } '`
131  [ "$ck2" ] || ck2=`cksum "$T/$artist/$title/$found" | awk ' { print $1 } '`
132  [ $ck1 -eq $ck2 ] && {
133  remlink "$T/$artist/$title/$found" "$i"
134  }
135  else
136  remlink "$T/$artist/$title/$found" "$i"
137  fi
138  }
139  [ "$found" ] || {
140  [ "$TELL" ] && continue
141  # Maybe the filename in iTunes has been changed. Search the SUMS
142  [ "$USE_SUM" ] && {
143  iname=`grep $ck1 "$ISUMS"`
144  [ "$iname" ] && {
145  b=`echo $iname | awk ' { for(i=3;i<NF;i++) printf "%s",$i OFS; if (NF) printf "%s",$NF; printf ORS}'`
146  b=`echo $b | sed -e "s/\.\///"`
147  remlink "$T/$b" "$i"
148  }
149  }
150  }
151  fi
152  done
153 }
