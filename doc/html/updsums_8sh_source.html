30 W="`pwd`"

32 TELLME=
33 FORCE=
34 
35 [ "$1" = "-d" ] && TELLME=1
36 [ "$1" = "-f" ] && FORCE=1
37 
38 [ -r "$SUMS" ] || {
39  echo "$SUMS does not exist."
40  echo "Creating - this may take a few minutes ..."
41  if [ "$TELLME" ]
42  then
43  find . -type f | while read i
44  do
45  echo "cksum $i"
46  done
47  else
48  rm -f /tmp/sums$$
49  touch /tmp/sums$$
50  find . -type f | while read i
51  do
52  cksum "$i" >> /tmp/sums$$
53  done
54  sort /tmp/sums$$ > "$SUMS"
55  rm -f /tmp/sums$$
56  fi
57  echo "Done."
58  exit 0
59 }
60 
61 # Remove entries in SUMS no longer in the tree
62 while read f
63 do
64  fnam=`echo "$f" | awk ' { for(i=3;i<NF;i++) printf "%s",$i OFS; if (NF) printf "%s",$NF; printf ORS}'`
65  [ -f "$fnam" ] || {
66  # Backslash the brackets in any filenames so grep doesn't choke
67  fixed=`echo "$f" | sed -e "s/\[/\\\\\[/g" -e "s/\]/\\\\\]/g"`
68  if [ "$TELLME" ]
69  then
70  echo "grep -v $f $SUMS"
71  else
72  grep -v "$fixed" "$SUMS" > /tmp/sums$$
73  cp /tmp/sums$$ "$SUMS"
74  rm -f /tmp/sums$$
75  fi
76  }
77 done < "$SUMS"
78 
79 # See if anything is new

81 then
82  newer=10
83 else
84  newer=`find . \( -type f -or -type d \) -newer "$SUMS" | wc -l`
85 fi
86 if [ $newer -lt 2 ]
87 then
88  echo "$SUMS is current"
89 else
90  # Add new files
91  find . -type f | while read f
92  do
93  fixed=`echo "$f" | sed -e "s/\[/\\\\\[/g" -e "s/\]/\\\\\]/g"`
94  grep "$fixed" "$SUMS" > /dev/null || {
95  if [ "$TELLME" ]
96  then
