57 ## @fn remlink()
58 ## @brief Remove file and replace it with a symbolic link
59 ## @param param1 Source file to use for symbolic link

61 remlink() {
62  printf "\nRemoving $2"
63  [ "$TELL" ] || rm -f "$2"
64  printf "\nLinking $1 to $2\n"
65  [ "$TELL" ] || ln -s "$1" "$2"
66 }

68 ## @fn FindAndLink()
69 ## @brief Recursively find and symlink duplicate files in current directory
70 ## @param none
71 FindAndLink() {
72  find . -type f | while read i
73  do
74  if [ -L "$i" ]
75  then
76  continue
77  else
78  b=`basename "$i"`
79  [ "$b" = "SUMS" ] && continue
80  # Backslash the brackets in any filenames so grep doesn't choke
81  fixed=`echo "$i" | sed -e "s/\[/\\\\\[/g" -e "s/\]/\\\\\]/g"`
82  ck1=`grep "$fixed" SUMS | awk ' { print $1 } '`
83  [ "$ck1" ] || ck1=`cksum "$i" | awk ' { print $1 } '`
84  # Could be in multiple Aperture, migrated Aperture, or Photos
85  # libraries, use the first one we find searching Photos then
86  # Aperture then migrated Aperture.
87  for lib in "$LIBS"/*.photoslibrary/Masters "$LIBS"/*.aplibrary/Masters "$LIBS"/*.migratedaplibrary/Masters
88  do
89  [ -d "$lib" ] && {
90  b=`grep "$ck1" "$lib/SUMS" | awk ' { for(i=3;i<NF;i++) printf "%s",$i OFS; if (NF) printf "%s",$NF; printf ORS}'`
91  ck2=`echo $b | sed -e "s/\.\///"`
92  [ "$ck2" ] || continue
93  remlink "$lib/$ck2" "$i"
94  break
95  }
96  done
97  fi
98  done
99 }
