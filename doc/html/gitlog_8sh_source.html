49 ## General git log options
50 ##
51 ## -p Show the patch introduced with each commit.
52 ## --stat Show statistics for files modified in each commit.
53 ## --shortstat Display only the changed/insertions/deletions line from
54 ## the --stat command.
55 ## --name-only Show the list of files modified after the commit information.
56 ## --name-status Show the list of files affected with added/modified/deleted
57 ## information as well.
58 ## --abbrev-commit Show only the first few characters of the SHA-1 checksum
59 ## instead of all 40.
60 ## --relative-date Display the date in a relative format (for example, “2 weeks
61 ## ago”) instead of using the full date format.
62 ## --graph Display an ASCII graph of the branch and merge history beside the
63 ## log output.
64 ## --pretty Show commits in an alternate format. Options include oneline, short,
65 ## full, fuller, and format (where you specify your own format).
66 ##
67 ## Options to filter output of git log
68 ##
69 ## -(n) Show only the last n commits
70 ## --since, --after Limit the commits to those made after the specified date.
71 ## --until, --before Limit the commits to those made before the specified date.
72 ## --author Only show commits in which the author entry matches the
73 ## specified string.
74 ## --committer Only show commits in which the committer entry matches the
75 ## specified string.
76 ## --grep Only show commits with a commit message containing the string

97 usage() {
98  printf "Usage: gitlog [-f] [-n] [-r] [-s days] [-u]\n"
99  printf "Where:\n\t-n indicates name-only output mode"
100  printf "\n\t-f specifies full log output (do not pretty format the output)"
101  printf "\n\t-r displays the log entries in reverse order.\n"
102  printf "\n\t-s # specifies log output for the last # days only"
103  printf "\n\t-u displays this usage message.\n"
104  printf "\nWould have executed the following command:"
105  if [ "$FULL" ]
106  then
107  printf "\n\tgit log $REVERSE $NAME_ONLY $SINCE\n\n"
108  else
109  printf "\n\tgit log $REVERSE --pretty=format:\"%%an, %%ad %%h :%%n %%s\" $NAME_ONLY $SINCE\n\n"
110  fi
111  exit 1
112 }
